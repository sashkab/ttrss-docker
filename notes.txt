# cody web server build from source

```bash
apk add --no-cache go git musl-dev \
&& mkdir $HOME/go \
&& export GOHOME=$HOME/go \
&& go get github.com/mholt/caddy/caddy \
&& go get github.com/caddyserver/builds \
&& cd $GOHOME/src/github.com/mholt/caddy/caddy \
&& cd git checkout v0.10.10 \
&& go run build.go
```
